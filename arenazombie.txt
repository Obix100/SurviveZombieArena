local function startFly()
    if fly then return end
    fly = true

    local Head = Character:WaitForChild("Head")

    Head.Anchored = true

    CFloop = RunService.Heartbeat:Connect(function(deltaTime)
        local camera = workspace.CurrentCamera
        local cameraCFrame = camera.CFrame

        local moveDirection = Humanoid.MoveDirection * (flySpeed * deltaTime)

        local headCFrame = Head.CFrame
        local cameraOffset = headCFrame:ToObjectSpace(cameraCFrame).Position

        cameraCFrame = cameraCFrame * CFrame.new(
            -cameraOffset.X,
            -cameraOffset.Y,
            -cameraOffset.Z + 1
        )

        local cameraPosition = cameraCFrame.Position
        local headPosition = headCFrame.Position

        local objectSpaceVelocity =
            CFrame.new(
                cameraPosition,
                Vector3.new(headPosition.X, cameraPosition.Y, headPosition.Z)
            ):VectorToObjectSpace(moveDirection)

        Head.CFrame =
            CFrame.new(headPosition) *
            (cameraCFrame - cameraPosition) *
            CFrame.new(objectSpaceVelocity)
    end)
end

local function stopFly()
    fly = false

    if CFloop then
        CFloop:Disconnect()
        CFloop = nil
    end

    local Head = Character:FindFirstChild("Head")

    if Head then
        Head.Anchored = false
    end
end

Movement:Toggle({

        if state then
            startFly()
        else

        end
    end

RunService.Heartbeat:Connect(function()
    local char = LocalPlayer.Character

    if not char then
        return
    end

    local hrp = char:FindFirstChild("HumanoidRootPart")

    if not hrp then
        return
    end

    if antiVoid and hrp.Position.Y < antiVoidHeight then
        hrp.CFrame = CFrame.new(
            hrp.Position.X,
            25,
            hrp.Position.Z
        )
    end

    if avoidZombies then
        local zombies = KAGetClosestZombies(1)

        if #zombies > 0 then
            local zombie = zombies[1]

            if zombie.Position then
                local dist = (zombie.Position - hrp.Position).Magnitude

                if dist <= 15 then
                    local randomOffset = Vector3.new(
                        math.random(-60, 60),
                        math.random(15, 30),
                        math.random(-60, 60)
                    )

                    hrp.CFrame = CFrame.new(hrp.Position + randomOffset)
                end
            end
        end
    end

    if orbitZombie then
        local now = tick()

        if now - lastOrbitSwitch >= 1 then
            local zombies = KAGetClosestZombies(10)

            if #zombies > 0 then
                orbitTarget = zombies[math.random(1, #zombies)]
                lastOrbitSwitch = now
            end
        end

        if orbitTarget and orbitTarget.Model and orbitTarget.Position then
            local angle = tick() * orbitSpeed

            local offset = Vector3.new(
                math.cos(angle) * orbitRadius,
                5,
                math.sin(angle) * orbitRadius
            )

            hrp.CFrame = CFrame.new(orbitTarget.Position + offset)
        end
    end
end)

if not antibanLoaded then
    antibanLoaded = true

    local mt = getrawmetatable(game)
    local oldNamecall = mt.__namecall
    local oldIndex = mt.__index
    local oldNewindex = mt.__newindex

    setreadonly(mt, false)

    mt.__namecall = newcclosure(function(self, ...)
        local method = getnamecallmethod()
        local args = {...}

        if antiKick then
            if method == "Kick" or method == "kick" then
                return nil
            end

            if method == "FireServer" or method == "InvokeServer" then
                local arg1 = tostring(args[1] or "")

                if arg1:lower():find("kick")
                or arg1:lower():find("ban")
                or arg1:lower():find("remove")
                or arg1:lower():find("punish")
                or arg1:lower():find("anticheat")
                or arg1:lower():find("sanity")
                or arg1:lower():find("validation")
                or arg1:lower():find("replication") then
                    return nil
                end
            end
        end

        return oldNamecall(self, ...)
    end)

    mt.__index = newcclosure(function(self, key)
        if antiKick and self == LocalPlayer and (key == "Kick" or key == "kick") then
            return nil
        end

        return oldIndex(self, key)
    end)

    mt.__newindex = newcclosure(function(self, key, value)
        if antiKick and self == LocalPlayer and (key == "Kick" or key == "kick") then
            return
        end

        return oldNewindex(self, key, value)
    end)

    setreadonly(mt, true)

    local oldFireServer = Instance.new("RemoteEvent").FireServer

    hookfunction(oldFireServer, function(self, ...)
        if antiKick then
            local arg1 = tostring(select(1, ...) or "")

            if arg1:lower():find("kick")
            or arg1:lower():find("ban")
            or arg1:lower():find("remove")
            or arg1:lower():find("punish")
            or arg1:lower():find("anticheat")
            or arg1:lower():find("sanity")
            or arg1:lower():find("validation")
            or arg1:lower():find("replication") then
                return
            end
        end

        return oldFireServer(self, ...)
    end)

                    hum.WalkSpeed = 16
